/*
Q1. Design a C program to find Fibonacci number using Dynamic Programming.
*/

#include <stdio.h>

int fibo(int n) {
    int dp[n + 1];

    dp[0] = 0;
    dp[1] = 1;

    for (int i = 2; i <= n; i++) {
        dp[i] = dp[i - 1] + dp[i - 2];
    }

    return dp[n];
}

int main() {
    int n;

    printf("Enter n: ");
    scanf("%d", &n);

    printf("Fibonacci value = %d\n", fibo(n));

    return 0;
}
